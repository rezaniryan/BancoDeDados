create table Cidades (
id int  primary key,
nome varchar(60)  not null,
populacao int
);

create table Alunos (
id	int  primary key,
nome varchar(60) not null,
data_nasc date,
cidade_id int,
foreign key(cidade_id) references Cidades(id)
);

 A.id as "RA",

 A.data_nasc as "Data de Nascimento"

 Alunos A left join

	on A.cidade_id = C.id
